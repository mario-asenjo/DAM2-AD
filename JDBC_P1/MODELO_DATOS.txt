CREATE TABLE PREGUNTAS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
	ENUNCIADO VARCHAR(200)
);

CREATE TABLE OPCION (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    TEXTO VARCHAR(100),
    ETIQUETA ENUM('A', 'B', 'C', 'D'),
    CORRECTA INT(1) DEFAULT 0,
    ID_PREGUNTA INT,
    FOREIGN KEY('ID_PREGUNTA') REFERENCES PREGUNTAS('ID')
    	ON DELETE CASCADE
    	ON UPDATE CASCADE
);
