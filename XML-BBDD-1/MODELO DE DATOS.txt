USE entrenadores_pokemon_hikari;

CREATE TABLE entrenadores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre        VARCHAR(100) NOT NULL,
    pueblo        VARCHAR(100),
    edad          INT
);

CREATE TABLE pokemons (
    id    INT AUTO_INCREMENT PRIMARY KEY,
    id_entrenador INT NOT NULL,
    nombre        VARCHAR(100) NOT NULL,
    especie       VARCHAR(100) NOT NULL,
    tipo          VARCHAR(30)  NOT NULL,
    nivel         INT NOT NULL,
    FOREIGN KEY (id_entrenador) REFERENCES entrenadores(id)
);

// Sirve para ambos!
DROP VIEW IF EXISTS entrenadores_pokemons;

CREATE VIEW entrenadores_pokemons AS
SELECT
    e.id            AS E_ID,
    e.nombre        AS E_NOMBRE,
    e.pueblo        AS E_PUEBLO,
    e.edad          AS E_EDAD,
    p.id            AS P_ID,
    p.nombre        AS P_NOMBRE,
    p.especie       AS P_ESPECIE,
    p.tipo          AS P_TIPO,
    p.nivel         AS P_NIVEL,
    p.id_entrenador AS P_ID_ENTRENADOR
FROM entrenadores e
LEFT JOIN pokemons p ON e.id = p.id_entrenador;
